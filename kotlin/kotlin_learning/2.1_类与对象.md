## 1. 基本类定义 —— 主构造函数、init 块、次构造函数

```kotlin
//1. 基本类：主构造函数 + init 块 + 次构造函数
class Person(val name: String, var age: Int) {
    // init 块：主构造函数执行后立即执行，可以有多个，按顺序执行
    init {
        require(age >= 0) { "年龄不能为负数" }
    }

    // 次构造函数：必须委托给主构造函数
    constructor(name: String) : this(name, 0)

    fun info() = "$name, $age 岁"
}

// 最简单的类
val p1 = Person("Alice", 25)
println(p1.info())

// 次构造函数：只传名字，年龄默认 0
val p2 = Person("Bob")
println(p2.info())
```

## 2. 属性（property） —— getter/setter、幕后字段 field

```kotlin
// 2. 属性：自定义 getter/setter、幕后字段
// Kotlin 的属性本质上是 getter + setter + 一个隐藏的存储字段
class Student(val name: String, score: Int) {
    // 带自定义 setter 的属性
    var score: Int = score
        set(value) {
            // field 是幕后字段，代表属性的实际存储值
            // 不能写 this.score = value，会无限递归
            field = value.coerceIn(0, 100)  // 限制在 0-100 之间
        }

    // 只读计算属性（没有幕后字段，每次访问都重新计算）
    val grade: String
        get() = when {
            score >= 90 -> "优秀"
            score >= 60 -> "及格"
            else -> "不及格"
        }

    // 带幕后字段的属性
    var loginCount: Int = 0
        private set  // setter 私有，外部只能读不能写

    fun login() {
        loginCount++
    }
}

val student = Student("Charlie", 90)
println("姓名: ${student.name}")
println("成绩: ${student.score}")
println("等级: ${student.grade}")  // 自定义 getter

student.score = 55
println("修改后成绩: ${student.score}")
println("修改后等级: ${student.grade}")

println("登录次数: ${student.loginCount}")
student.login()
student.login()
println("登录后次数: ${student.loginCount}")
```

## 3. data class —— 自动生成 equals/hashCode/toString/copy

```kotlin
// 3. data class：编译器自动生成 equals/hashCode/toString/copy/componentN
data class User(
    val name: String,
    val age: Int,
    val email: String
)
// 注意：data class 至少要有一个主构造函数参数
// 只有主构造函数中的属性参与 equals/hashCode/toString

val user1 = User("Alice", 25, "alice@example.com")
val user2 = User("Alice", 25, "alice@example.com")
val user3 = User("Bob", 30, "bob@example.com")

// 自动生成 toString
println(user1)  // User(name=Alice, age=25, email=alice@example.com)

// 自动生成 equals（比较内容而非引用）
println("user1 == user2: ${user1 == user2}")  // true
println("user1 == user3: ${user1 == user3}")  // false

// copy：复制对象并修改部分属性（不可变数据的常用操作）
val user4 = user1.copy(age = 26)
println("copy 后: $user4")  // name 和 email 不变，age 变了

// 解构声明
val (name, age, email) = user1
println("解构: name=$name, age=$age, email=$email")
```

## 4. sealed class —— 限定子类，常用于 UI 状态建模

```kotlin
// 4. sealed class：密封类，子类必须定义在同一个文件中（Kotlin 1.5+ 同一个包内也行）
// Android 中最常见的用法：表示 UI 状态
sealed class NetworkResult {
    // object：没有数据的状态用 object（单例）
    object Loading : NetworkResult()

    // data class：有数据的状态用 data class
    data class Success(val data: List<String>) : NetworkResult()
    data class Error(val message: String) : NetworkResult()
}

// sealed class 配合 when 使用，编译器会检查是否覆盖了所有子类
fun showResult(result: NetworkResult) {
    when (result) {
        is NetworkResult.Loading -> println("加载中...")
        is NetworkResult.Success -> println("成功: ${result.data}")  // 智能转换
        is NetworkResult.Error -> println("失败: ${result.message}")
        // 不需要 else，因为编译器知道所有子类都覆盖了
    }
}

// 模拟网络请求的三种状态
val loading: NetworkResult = NetworkResult.Loading
val success: NetworkResult = NetworkResult.Success(listOf("数据1", "数据2"))
val error: NetworkResult = NetworkResult.Error("网络超时")

showResult(loading)
showResult(success)
showResult(error)
```

## 5. object 单例 —— 天然的单例模式

```kotlin
// 5. object：单例，天然线程安全
object AppConfig {
    var appName: String = ""
    var debug: Boolean = false

    fun printConfig() {
        println("App: $appName, Debug: $debug")
    }
}

// 直接用类名访问，全局只有一个实例
AppConfig.appName = "MyApp"
AppConfig.debug = true
AppConfig.printConfig()

// 两次引用是同一个对象
val config1 = AppConfig
val config2 = AppConfig
println("是同一个实例: ${config1 === config2}")  // true
```

## 6. companion object —— 类似 Java 的 static

```kotlin
// 6. companion object：伴生对象，类似 Java 的 static
data class HttpClient(val baseUrl: String, val timeout: Int) {
    companion object {
        const val DEFAULT_TIMEOUT = 30

        // 工厂方法（Android 中很常见的模式）
        fun create(baseUrl: String, timeout: Int = DEFAULT_TIMEOUT): HttpClient {
            return HttpClient(baseUrl, timeout)
        }
    }
}

// 通过类名直接调用（像 Java 的 static 方法）
val http = HttpClient.create("https://api.example.com")
println(http)
println("默认超时: ${HttpClient.DEFAULT_TIMEOUT}")

// Android 中常见用法：Fragment.newInstance()
// val fragment = MyFragment.newInstance("参数")
```

## 7. 枚举类 enum class

```kotlin
// 7. 枚举类：可以有属性和方法
enum class DayOfWeek(val cnName: String, val isWorkday: Boolean) {
    MONDAY("星期一", true),
    TUESDAY("星期二", true),
    WEDNESDAY("星期三", true),
    THURSDAY("星期四", true),
    FRIDAY("星期五", true),
    SATURDAY("星期六", false),
    SUNDAY("星期日", false);

    // 枚举也可以有方法
    fun emoji() = if (isWorkday) "💼" else "🎉"
}

val today = DayOfWeek.WEDNESDAY
println("今天: ${today.cnName}, 是否工作日: ${today.isWorkday}")

// 遍历所有枚举值
DayOfWeek.entries.forEach { day ->
    println("  ${day.name} -> ${day.cnName}")
}

// 配合 when 使用（编译器会检查是否覆盖所有值）
val plan = when (today) {
    DayOfWeek.MONDAY -> "开周会"
    DayOfWeek.FRIDAY -> "写周报"
    DayOfWeek.SATURDAY, DayOfWeek.SUNDAY -> "休息"
    else -> "正常上班"
}
println("今日安排: $plan")
```

## 8. 嵌套类与内部类（nested vs inner）

```kotlin
// 8. 嵌套类 vs 内部类
class Outer(private val data: String = "默认") {
    // 嵌套类（默认）：不持有外部类引用，类似 Java 的 static inner class
    class Nested {
        fun show() {
            println("我是嵌套类，不能访问外部类的成员")
            // println(data)  // 编译错误！
        }
    }

    // 内部类（加 inner 关键字）：持有外部类引用
    inner class Inner {
        fun show() {
            println("我是内部类，可以访问外部类的 data = $data")
        }
    }
}

// 嵌套类：不持有外部类引用（默认行为）
val nested = Outer.Nested()
nested.show()

// 内部类：持有外部类引用，可以访问外部类成员
val outer = Outer("外部数据")
val inner = outer.Inner()
inner.show()
```
